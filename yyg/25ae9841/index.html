--nameеҗҺеЎ«зҡ„mysql-1жҳҜиҝҗиЎҢзҡ„mysqlе®№еҷЁзҡ„еҗҚз§°пјҢд№ҹеҸҜиҮӘиЎҢжӣҝжҚў

docker run -d -p 3306:3306 вҖ”privileged=true -v /jiamian/mysql/log:/var/log/mysql

-v /jiamian/mysql/data:/var/lib/mysql -v /jiamian/mysql/conf:/etc/mysql/conf.d

-e MYSQL_ROOT_PASSWORD=123456 вҖ”name mysql-1 mysql

иҝҗиЎҢdocker psе‘Ҫд»ӨпјҢеҸҜжҹҘзңӢеҲ°mysqlжӯЈеңЁиҝҗиЎҢ
