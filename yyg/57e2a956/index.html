任务四：安装高可用组件

部署高可用集群需要安装keepalived和haproxy，实现**master节点高可用，在各master节点操作**

  1. 安装keepalived与haproxy
  1. 备份keepalived与haproxy配置文件

  1. 修改各master节点的/etc/keepalived/keepalived.conf文件

    1. future-k8s-master0

    2. ```Shell
      echo ‘
      global_defs {
      router_id k8s
      }

      vrrp_script check_haproxy {

      script "killall -0 haproxy"
interval 3
weight -2
fall 10
rise 2

      }

      vrrp_instance VI_1 {

      state MASTER  #主节点 则为MASTER ,其他则为 BACKUP
interface ens192  #网卡名称
virtual_router_id 51
priority 250   #优先级
nopreempt   #设置非抢占模式
advert_int 1
authentication {
    auth_type PASS
    auth_pass ceb1b3ec013d66163d6ab
}
virtual_ipaddress {
    10.10.1.83/24   #虚拟ip
}
track_script {
    check_haproxy
}

      }
      ‘ > /etc/keepalived/keepalived.conf
      cat /etc/keepalived/keepalived.conf

    3. future-k8s-master2

    4. ```Shell
      echo ‘
      global_defs {
      router_id k8s
      }

      vrrp_script check_haproxy {

      script "killall -0 haproxy"
interval 3
weight -2
fall 10
rise 2

      }

      vrrp_instance VI_1 {

      state BACKUP  #主节点 则为MASTER ,其他则为 BACKUP
interface ens192  #网卡名称
virtual_router_id 51
priority 150   #优先级
nopreempt   #设置非抢占模式
advert_int 1
authentication {
    auth_type PASS
    auth_pass ceb1b3ec013d66163d6ab
}
virtual_ipaddress {
    10.10.1.83/24   #虚拟ip
}
track_script {
    check_haproxy
}

      }
      ‘ > /etc/keepalived/keepalived.conf
      cat /etc/keepalived/keepalived.conf
